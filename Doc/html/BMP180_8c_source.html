141HAL_StatusTypeDef BMP180_ReadSensorRaw(BMP180_HandleTypeDef *hbmp, BMP180_OversamplingTypeDef precision, uint16_t *UT, uint32_t *UP) {
142 BMP180_ControlRegisterTypeDef controlRegister;
143 HAL_StatusTypeDef result;
144
145 HAL_StatusTypeDef waitForConversion() {
146 do {
147 if ((result = BMP180_ReadRegister(hbmp, 0xF4, (uint8_t *)&controlRegister, sizeof(controlRegister))))
148 return result;
149 else if (controlRegister.sco == BMP180_CONVERSION_RUNNING)
150 HAL_Delay(2);
151 }
152 while (controlRegister.sco == BMP180_CONVERSION_RUNNING);
153 return result;
154 }
155
156 controlRegister.measurement = BMP180_MEASUREMENT_TEMPERATURE;
157 controlRegister.oss = BMP180_OVERSAMPLING_0;
158 controlRegister.sco = BMP180_CONVERSION_RUNNING;
159 if ((result = BMP180_WriteRegister(hbmp, 0xF4, (uint8_t *)&controlRegister, sizeof(controlRegister))))
160 goto HAL_BMP180_ReadSensor_Exit;
161
162 if ((result = waitForConversion()))
163 goto HAL_BMP180_ReadSensor_Exit;
164
165 uint16_t UTbuffer = 0;
166 if ((result = BMP180_ReadRegister(hbmp, 0xF6, (uint8_t *)&UTbuffer, sizeof(short))))
167 goto HAL_BMP180_ReadSensor_Exit;
168 *UT = __builtin_bswap16(UTbuffer);
169
170 controlRegister.measurement = BMP180_MEASUREMENT_PRESSURE;
171 controlRegister.oss = precision;
172 controlRegister.sco = BMP180_CONVERSION_RUNNING;
173 if ((result = BMP180_WriteRegister(hbmp, 0xF4, (uint8_t *)&controlRegister, sizeof(controlRegister))))
174 goto HAL_BMP180_ReadSensor_Exit;
175
176 if (waitForConversion() != HAL_OK)
177 goto HAL_BMP180_ReadSensor_Exit;
178
179 uint32_t UPbuffer = 0;
180 if ((result = BMP180_ReadRegister(hbmp, 0xF6, (uint8_t *)&UPbuffer, 3)))
181 goto HAL_BMP180_ReadSensor_Exit;
182 *UP = (__builtin_bswap32(UPbuffer) >> (16 - controlRegister.oss));
183HAL_BMP180_ReadSensor_Exit:
184 return result;
185}

193long BMP180_CompensateTemp(BMP180_HandleTypeDef *hbmp, long UT) {
194 long X1, X2, B5;
195
196 X1 = (((long)UT - (long)hbmp->CalibrationData.AC6) * (long)hbmp->CalibrationData.AC5) >> 15;
197 X2 = ((long)hbmp->CalibrationData.MC << 11)/(X1 + hbmp->CalibrationData.MD);
198 B5 = X1 + X2;
199 return ((B5 + 8) >> 4);

211long HAL_BMP180_CompensatePressure(BMP180_HandleTypeDef *hbmp, BMP180_OversamplingTypeDef precision, long UT, long UP) {
212 long X1, X2, X3;
213 long B3, B5, B6;
214 unsigned long B4, B7;
215 long p;
216
217 X1 = (((long)UT - (long)hbmp->CalibrationData.AC6) * (long)hbmp->CalibrationData.AC5) >> 15;
218 X2 = ((long)hbmp->CalibrationData.MC << 11)/(X1 + hbmp->CalibrationData.MD);
219 B5 = X1 + X2;
220
221 B6 = B5 - 4000;
222
223 X1 = (hbmp->CalibrationData.B2 * (B6 * B6) >> 12) >> 11;
224 X2 = (hbmp->CalibrationData.AC2 * B6) >> 11;
225 X3 = X1 + X2;
226 B3 = (((((long)hbmp->CalibrationData.AC1) * 4 + X3) << precision) + 2) >> 2;
227
228 X1 = (hbmp->CalibrationData.AC3 * B6) >> 13;
229 X2 = (hbmp->CalibrationData.B1 * ((B6 * B6) >> 12)) >> 16;
230 X3 = ((X1 + X2) + 2) >> 2;
231 B4 = (hbmp->CalibrationData.AC4 * (unsigned long)(X3 + 32768)) >> 15;
232
233 B7 = ((unsigned long)(UP - B3) * (50000 >> precision));
234
235 if (B7 < 0x80000000)
236 p = (B7 << 1) / B4;
237 else
238 p = (B7 / B4) << 1;
239
240 X1 = (p >> 8) * (p >> 8);
241 X1 = (X1 * 3038) >> 16;
242 X2 = (-7357 * p) >> 16;
243 p += (X1 + X2 + 3791) >> 4;
244
245 return p;
246}
